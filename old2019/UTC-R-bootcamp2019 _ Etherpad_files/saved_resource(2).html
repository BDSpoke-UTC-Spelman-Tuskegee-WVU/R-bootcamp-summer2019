 #permutation of pairs, and their difference in YPD
 Nsims = 10; #number of permutations
 permutated.diff.YPD = numeric( Nsims ); #empty vector to store calculations

 # Now do N simulations via for loop
 for( i in 1:Nsims ) {
  
   new.pairs.YPD = ms02_singlerun(pairs ) #generate a new MS02 random network
   new.pairs.YPD = new.pairs.YPD[,1:2] #reformating into two-columns
   names(new.pairs.YPD) = c("ORF1", "ORF2")

   #calculate delta.YPD for one random network
